@startuml
actor "<u>:Admin" as admin
participant "<u>:Terminal" as terminal
participant "<u>:InstructorCollection" as instructorCollection
participant "<u>:ClientCollection" as clientCollection
collections "<u>:Instructor" as instructors
collections "<u>:Client" as clients

admin -> terminal : requestToDeleteUser()
activate terminal
activate admin
terminal --> admin : requestUserTypeToDelete
deactivate terminal
deactivate admin

admin -> terminal : specifyUserType(userType)
activate terminal
activate admin
alt Instructor
    terminal -> instructorCollection : getInstructors()
    activate terminal
    activate instructorCollection
    instructorCollection -> instructors : getAll()
    instructors -> instructorCollection : instructors
    instructorCollection -> terminal : instructors
    deactivate terminal
    deactivate instructorCollection
else Client
    terminal -> clientCollection : getClients()
    activate terminal
    activate clientCollection
    clientCollection -> clients : getAll()
    clients -> clientCollection : clients
    clientCollection -> terminal : clients
    deactivate terminal
    deactivate clientCollection
end
terminal --> admin : displayUsers
deactivate terminal
deactivate admin

admin -> terminal : inputId(id)
activate admin
activate terminal

terminal -> admin : requestConfirmation
deactivate admin
deactivate terminal

admin -> terminal : confirmDeletion()
activate admin
activate terminal

alt Instructor
    terminal -> instructorCollection : removeInstructor(id)
    activate terminal
    activate instructorCollection
    instructorCollection -> instructors : remove(id)
    alt Successful Database Operation
        instructors -> instructorCollection : success
        instructorCollection --> terminal : success
    else Failed Database Operation
        instructors -> instructorCollection : failure
        instructorCollection --> terminal : failure
    deactivate instructorCollection
    deactivate terminal
    end
else Client
    terminal -> clientCollection : removeClient(id)
    activate terminal
    activate clientCollection
    clientCollection -> clients : remove(id)
    alt Successful Database Operation
        clients -> clientCollection : success
        clientCollection --> terminal : success
    else Failed Database Operation
        clients -> clientCollection : failure
        clientCollection --> terminal : failure
    deactivate terminal
    deactivate clientCollection
    end
end

terminal --> admin : response
deactivate admin
deactivate terminal
@enduml
